def shuffle_pixels(images, percent=0.1):
    shuffled_images = np.zeros_like(images)
    for i, image in enumerate(images):
        # Flatten the image
        flattened_image = image.reshape(-1, 3)
        # Get number of pixels to shuffle
        num_pixels = int(percent * flattened_image.shape[0])
        # Randomly select pixels to shuffle
        indices = np.random.choice(flattened_image.shape[0], num_pixels, replace=False)
        # Shuffle selected pixels
        np.random.shuffle(flattened_image[indices])
        # Reshape back to original shape
        shuffled_images[i] = flattened_image.reshape(32, 32, 3)
    return shuffled_images
